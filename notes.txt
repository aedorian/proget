typedef struct { int x; int y; } couple;
struct transform
  couple pos
  couple dir

struct game
  balle* balles []   # ou deux listes?

  ennemi* ennemis []
  taille n utilisée pour les 3

struct ennemi
  int vie
  hitbox
  transform transform

struct joueur
  transform transform
  int vie
  arme* arme;    # arme courante
  couple hitbox

struct balle
  hitbox hitbox
  transform transform
  int/float damage
  int estJoueur?
  int/float vitesse
  MLV_Image image

enum TYPE_TIR { SINGLE, CONTINOUS };         # mode activation

  MODE_TIR mode_tir
